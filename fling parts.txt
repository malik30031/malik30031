-- checks


local function hasDoorInAncestry(instance)


	while instance do


		if string.lower(instance.Name):find("door") then
			return true
		end


		instance=instance.Parent


	end


	return false


end


local function hasBlockedInAncestry(part)


	local blocked={"motor","car","sit","swing"}


	while part do


		local name=string.lower(part.Name)


		for _,w in pairs(blocked) do
			if name:find(w) then
				return true
			end
		end


		part=part.Parent


	end


	return false


end


-- التحقق إذا كان الجزء تابع لشخصية لاعب


local function isPlayerCharacterPart(part)


	local model=part:FindFirstAncestorOfClass("Model")


	if model then


		for _,plr in pairs(Players:GetPlayers()) do


			if plr.Character and model==plr.Character then
				return true
			end


		end


	end


	return false


end


-- التحقق إذا كان الجزء تابع لأي Tool


local function isToolPart(part)
	return part:FindFirstAncestorOfClass("Tool")~=nil
end


-- table


local controlledParts={}


-- force part


local function ForcePart(v)


	if v:IsA("BasePart")
	and not v.Anchored
	and v.Name~="Handle"
	and not hasDoorInAncestry(v)
	and not hasBlockedInAncestry(v)
	and not isPlayerCharacterPart(v)
	and not isToolPart(v)
	and not controlledParts[v]
	then


		v.CanCollide=false


		for _,x in ipairs(v:GetChildren()) do
			if x:IsA("BodyMover") or x:IsA("RocketPropulsion") then
				x:Destroy()
			end
		end


		local Torque=Instance.new("Torque")
		Torque.Torque=Vector3.new(100000,100000,100000)
		Torque.Parent=v


		local AlignPosition=Instance.new("AlignPosition")
		AlignPosition.MaxForce=math.huge
		AlignPosition.MaxVelocity=math.huge
		AlignPosition.Responsiveness=200
		AlignPosition.Parent=v


		local Attachment2=Instance.new("Attachment",v)


		Torque.Attachment0=Attachment2


		AlignPosition.Attachment0=Attachment2
		AlignPosition.Attachment1=Attachment1


		controlledParts[v]=true


	end


end


-- blackhole


local blackHoleActive=false
local DescendantAddedConnection


local function toggleBlackHole()


	blackHoleActive=not blackHoleActive


	if blackHoleActive then


		Button.Text="Bring Parts | On"


		for _,v in ipairs(Workspace:GetDescendants()) do
			ForcePart(v)
		end


		DescendantAddedConnection=Workspace.DescendantAdded:Connect(function(v)


			if blackHoleActive then
				ForcePart(v)
			end


		end)


	else


		Button.Text="Bring Parts | Off"


		if DescendantAddedConnection then
			DescendantAddedConnection:Disconnect()
		end


	end


end


-- player search


local function getPlayer(name)


	local lower=string.lower(name)


	for _,p in pairs(Players:GetPlayers()) do


		if string.find(string.lower(p.Name),lower) or string.find(string.lower(p.DisplayName),lower) then
			return p
		end


	end


end


local player=nil


Box.FocusLost:Connect(function(enter)


	if enter then


		player=getPlayer(Box.Text)


		if player then
			Box.Text=player.Name
		end


	end


end)


-- follow


local function followPlayer(player)


	task.spawn(function()


		while blackHoleActive do


			if player and player.Character then


				local hrp=player.Character:FindFirstChild("HumanoidRootPart")


				if hrp then
					Attachment1.WorldCFrame=hrp.CFrame
				end


			end


			RunService.RenderStepped:Wait()


		end


	end)


end


Button.MouseButton1Click:Connect(function()


	if player then


		toggleBlackHole()


		if blackHoleActive then
			followPlayer(player)
		end


	end


end)
